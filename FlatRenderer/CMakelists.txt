cmake_minimum_required(VERSION 3.0.0)
project(flat2RendererTest VERSION 0.1.0)

add_library(renderer STATIC)
target_compile_features(renderer PUBLIC cxx_std_20)
target_sources(renderer PRIVATE renderer.hpp gl_renderer.hpp gl_renderer.cpp ${CMAKE_CURRENT_LIST_DIR}/../3rdparty/glad/src/glad.c)
target_include_directories(renderer PUBLIC 
    ${CMAKE_CURRENT_LIST_DIR}/../3rdparty/glad/include
    ${CMAKE_CURRENT_LIST_DIR}/../3rdparty/glm
    ${CMAKE_CURRENT_LIST_DIR}/../3rdparty/stb
    ${CMAKE_CURRENT_LIST_DIR}/../FlatWindow/
    )

add_executable(renderer_test)
add_dependencies(renderer_test window)
target_compile_features(renderer_test PUBLIC cxx_std_20)
target_sources(renderer_test PRIVATE test.cpp)
target_link_libraries(renderer_test PUBLIC renderer window)
