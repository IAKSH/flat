cmake_minimum_required(VERSION 3.0.0)
project(flat2CollisionTest VERSION 0.1.0)

add_library(collision STATIC)
target_compile_features(collision PUBLIC cxx_std_20)
target_sources(collision PRIVATE flat_gjk/gjk.hpp flat_gjk/gjk.cpp)
target_include_directories(collision PUBLIC ${CMAKE_CURRENT_LIST_DIR}/../3rdparty/glm)

add_executable(collision_test)
add_dependencies(collision_test window)
target_include_directories(collision_test PUBLIC ${CMAKE_CURRENT_LIST_DIR}/../FlatWindow)
target_compile_features(collision_test PRIVATE cxx_std_20)
target_sources(collision_test PRIVATE test.cpp)
target_link_libraries(collision_test PRIVATE collision window)
