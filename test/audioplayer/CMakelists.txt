cmake_minimum_required(VERSION 3.0.0)
project(flat3 VERSION 0.1.0)

include(CTest)
enable_testing()

add_executable(gtestprogram_audioplayer)
add_dependencies(gtestprogram_audioplayer gtest glfw)
target_compile_features(gtestprogram_audioplayer PRIVATE cxx_std_20)
target_sources(gtestprogram_audioplayer PRIVATE 
    test.cpp 
    initialized.hpp 
    abstruct_api.hpp 
    alsoft_imp.hpp 
    static_type_check.hpp 
    )

target_include_directories(gtestprogram_audioplayer PUBLIC 
    ${CMAKE_CURRENT_LIST_DIR}/../../3rdparty/googletest/googletest/include
    )

target_link_libraries(gtestprogram_audioplayer PRIVATE gtest)

set(CPACK_PROJECT_NAME ${PROJECT_NAME})
set(CPACK_PROJECT_VERSION ${PROJECT_VERSION})
include(CPack)
