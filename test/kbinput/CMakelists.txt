cmake_minimum_required(VERSION 3.0.0)
project(flat3 VERSION 0.1.0)

include(CTest)
enable_testing()

add_executable(gtestprogram_kbinput)
add_dependencies(gtestprogram_kbinput gtest glfw)
target_compile_features(gtestprogram_kbinput PRIVATE cxx_std_20)
target_sources(gtestprogram_kbinput PRIVATE 
    test.cpp 
    initialized.hpp 
    abstruct_api.hpp 
    glfw_imp.hpp 
    static_type_check.hpp 
    ${CMAKE_CURRENT_LIST_DIR}/../../3rdparty/glad/src/glad.c
    )

target_include_directories(gtestprogram_kbinput PUBLIC 
    ${CMAKE_CURRENT_LIST_DIR}/../../3rdparty/googletest/googletest/include
    ${CMAKE_CURRENT_LIST_DIR}/../../3rdparty/glfw/include
    ${CMAKE_CURRENT_LIST_DIR}/../../3rdparty/glad/include
    )

target_link_libraries(gtestprogram_kbinput PRIVATE gtest glfw)

set(CPACK_PROJECT_NAME ${PROJECT_NAME})
set(CPACK_PROJECT_VERSION ${PROJECT_VERSION})
include(CPack)
